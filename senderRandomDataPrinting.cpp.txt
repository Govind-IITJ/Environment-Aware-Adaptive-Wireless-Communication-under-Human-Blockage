#include <esp_now.h>
#include <WiFi.h>

// ================= STRUCTURE =================
typedef struct struct_message {
  char name[32];
  int32_t number;
  float value;
  bool flag;
  uint32_t timestamp;   // milliseconds since boot
} struct_message;

struct_message myData;

//  RECEIVER MAC address
uint8_t broadcastAddress[] = {0x78, 0x1C, 0x3C, 0xA6, 0x1F, 0x88};

esp_now_peer_info_t peerInfo;

// ================= CALLBACK =================
void OnDataSent(const wifi_tx_info_t *info, esp_now_send_status_t status) {
  Serial.print("Last Packet Send Status: ");
  Serial.println(status == ESP_NOW_SEND_SUCCESS ? "Delivery Success" : "Delivery Fail");
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);
  delay(1000);

  WiFi.mode(WIFI_STA);   // Required for ESP-NOW

  // Initialize ESP-NOW
  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  // Register send callback
  esp_now_register_send_cb(OnDataSent);

  // Set peer information
  memset(&peerInfo, 0, sizeof(peerInfo));
  memcpy(peerInfo.peer_addr, broadcastAddress, 6);
  peerInfo.channel = 0;        // Same WiFi channel
  peerInfo.encrypt = false;    // No encryption

  // Add peer
  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }

  Serial.println("ESP-NOW Sender Ready");
}

// ================= LOOP =================
void loop() {

  // Generate data
  int32_t randomNumber = random(1, 100);
  float computedValue = 1.5 * randomNumber;
  static bool toggleFlag = false;
  toggleFlag = !toggleFlag;

  // Fill structure
  strcpy(myData.name, "Yo_Yo");
  myData.number = randomNumber;
  myData.value = computedValue;
  myData.flag = toggleFlag;
  myData.timestamp = millis();   // Timestamp added here

  // Send data
  esp_err_t result = esp_now_send(
      broadcastAddress,
      (uint8_t *) &myData,
      sizeof(myData)
  );

  if (result == ESP_OK) {
    Serial.println("Sending confirmed");
  } else {
    Serial.println("Sending error");
  }

  // Debug print locally
  Serial.print("Sent -> ");
  Serial.print("Name: "); Serial.print(myData.name);
  Serial.print(" | Number: "); Serial.print(myData.number);
  Serial.print(" | Value: "); Serial.print(myData.value);
  Serial.print(" | Flag: "); Serial.print(myData.flag);
  Serial.print(" | Timestamp: "); Serial.println(myData.timestamp);

  delay(2000);
}
